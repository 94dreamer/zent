import { Popup, PopupContent, PopupPosition, useClickPopup } from 'zent';

function App() {
	const [anchorProps, popup] = useClickPopup(PopupPosition.rightCenter);

	return (
		<span {...anchorProps}>
			Popup
			<PopupContent popup={popup}>Hello</PopupContent>
		</span>
	);
}

ReactDOM.render(<App />, mountNode);
